 48 /**
 49  * Color in green the passed button.
 50  * @param button An HTML button
 51  */
 52 function colorButton(button) {
 53 
 54     button.css("background-color", "#28A745");
 55 
 56     button.css("color", "white");
 57 
 58     button.data('clicked', true);
 59 
 60 }
 61 
 62 /**
 63  * Un-color in grey the passed button.
 64  * @param button An HTML button
 65  */
 66 function uncolorButton(button) {
 67 
 68     button.css("background-color", "white");
 69 
 70     button.css("color", "black");
 71 
 72     button.data('clicked', false);
 73 
 74 }

102 $(".btnStar").on('click', function() {

220 /**
221  * Add br in the menu.
222  */
223 function addBr(number) {
224 
225     // If there is no br yet
226     //if(document.getElementById("brType1").length === 0) {
227     if (number === 1) {
228 
229         var containerType = document.getElementById("typeContainer");
230 
231         var containerPrice = document.getElementById("currencyContainer");
232 
233         var containerStar = document.getElementById("starContainer");
234 
235         // We add the breaklines to the DOM via inserting them
236         var i = 1;
237 
238         // While we get a button, we add a breakline before its
239         while(containerType.getElementsByTagName("button")[i] && i < 3) {
240 
241             var newBrType = document.createElement("br"); // We create a new breakline
242 
243             newBrType.setAttribute("id", "brType"+i);
244 
245             containerType.insertBefore(newBrType, containerType.getElementsByTagName("button")[i]);
246 
247             i++;
248 
249         }
250 
251         i = 1;
252 
253         while(containerPrice.getElementsByTagName("button")[i] && i < 4) {
254 
255             var newBrPrice = document.createElement("br");
256 
257             newBrPrice.setAttribute("id", "brPrice"+i);
258 
259             containerPrice.insertBefore(newBrPrice, containerPrice.getElementsByTagName("button")[i]);
260 
261             i++;
262 
263         }
264 
265         i = 1;
266 
267         while(containerStar.getElementsByTagName("button")[i] && i < 5) {
268 
269             var newBrStar= document.createElement("br");
270 
271             newBrStar.setAttribute("id", "brStar"+i);
272 
273             containerStar.insertBefore(newBrStar, containerStar.getElementsByTagName("button")[i]);
274 
275             i++;
276 
277         }
278 
279         var newBrDay= document.createElement("br");
280 
281         newBrDay.setAttribute("id", "brDay");
282 
283         // We must insert in the parent node
284         var day = document.getElementById("labelDay").parentNode;
285 
286         day.insertBefore(newBrDay, document.getElementById("labelDay"));
287 
288         // Shrink the drop-down menu
289         document.getElementsByClassName("dropdown-menu").item(0).style.width = "300px";
290 
291         // Resize the search bar
292         document.getElementById("textSearch").style.width = "250px";
293 
294     }
295 
296 }
297 
298 /**
299  * Remove br in the menu.
300  */
301 function removeBr() {
302 
303     // For loops
304     var i;
305 
306     // We remove the breaklines if they exist
307     for (i = 1 ; i < 3 ; i++) {
308 
309         if(document.getElementById("brType"+i) !== undefined && document.getElementById("brType"+i) !== null) {
310 
311             document.getElementById("brType"+i).remove();
312 
313         }
314 
315     }
316 
317     for (i = 1 ; i < 4 ; i++) {
318 
319         if(document.getElementById("brPrice"+i) !== undefined && document.getElementById("brPrice"+i) !== null) {
320 
321             document.getElementById("brPrice"+i).remove();
322 
323         }
324 
325     }
326 
327     for (i = 1 ; i < 5 ; i++) {
328 
329         if(document.getElementById("brStar"+i) !== undefined && document.getElementById("brStar"+i) !== null) {
330 
331             document.getElementById("brStar"+i).remove();
332 
333         }
334 
335     }
336 
337     if(document.getElementById("brDay") !== undefined && document.getElementById("brDay") !== null) {
338 
339         document.getElementById("brDay").remove();
340 
341     }
342 
343     // Expand the dropdown menu
344     document.getElementsByClassName("dropdown-menu").item(0).style.width = "460px";
345 
346 }
